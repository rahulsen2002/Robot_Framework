"""A module to handle different character widths on the console. 
 
Some East Asian characters have width of two on console, and combining 
characters themselves take no extra space. 
 
For more details about East Asian characters and the associated problems see: 
https://github.com/robotframework/robotframework/issues/604 
"""


def get_char_width(char):
    char = ord(char)
    if _char_in_map(char, COMBINING_CHARS):
        return 0
    if _char_in_map(char, EAST_ASIAN_WILD_CHARS):
        return 2
    return 1


def _char_in_map(char, map):
    for begin, end in map:
        if char < begin:
            return False
        if char <= end:
            return True
    return False
