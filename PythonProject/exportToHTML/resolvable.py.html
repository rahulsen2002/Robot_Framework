from robot.errors import DataError


class Resolvable:

    def resolve(self, variables):
        raise NotImplementedError

    def report_error(self, error):
        raise DataError(error)


class GlobalVariableValue(Resolvable):

    def __init__(self, value):
        self.value = value

    def resolve(self, variables):
        return self.value
